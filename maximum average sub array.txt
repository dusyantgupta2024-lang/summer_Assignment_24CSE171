double findMaxAverage(int* nums, int numsSize, int k) {
    long sum = 0;

    // First window of size k
    for (int i = 0; i < k; i++) {
        sum += nums[i];
    }

    long maxSum = sum;
    for (int i = k; i < numsSize; i++) {
        sum = sum + nums[i] - nums[i - k];

        if (sum > maxSum) {
            maxSum = sum;
        }
    }

    return (double)maxSum / k;
}
